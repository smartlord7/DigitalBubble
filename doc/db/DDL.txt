CREATE TABLE product (
	id		 BIGINT,
	version	 BIGINT,
	name		 VARCHAR(512) NOT NULL,
	price		 FLOAT(8) NOT NULL,
	stock		 INTEGER NOT NULL,
	description	 VARCHAR(512),
	seller_user_id BIGINT NOT NULL,
	PRIMARY KEY(id,version)
);

CREATE TABLE computer (
	cpu		 VARCHAR(512) NOT NULL,
	gpu		 VARCHAR(512),
	product_id	 BIGINT,
	product_version BIGINT,
	PRIMARY KEY(product_id,product_version)
);

CREATE TABLE user (
	id		 BIGINT,
	name	 VARCHAR(512) UNIQUE NOT NULL,
	firstname	 VARCHAR(512) NOT NULL,
	email	 VARCHAR(512) UNIQUE NOT NULL,
	tin		 VARCHAR(512),
	lastname	 VARCHAR(512) NOT NULL,
	phonenumber	 VARCHAR(512) UNIQUE NOT NULL,
	passwordhash VARCHAR(256) NOT NULL,
	houseno	 INTEGER NOT NULL,
	streetname	 VARCHAR(512) NOT NULL,
	city	 VARCHAR(512) NOT NULL,
	state	 VARCHAR(512) NOT NULL,
	zipcode	 VARCHAR(64) NOT NULL,
	PRIMARY KEY(id)
);

CREATE TABLE television (
	size		 INTEGER NOT NULL,
	technology	 VARCHAR(512) NOT NULL,
	product_id	 BIGINT,
	product_version BIGINT,
	PRIMARY KEY(product_id,product_version)
);

CREATE TABLE smartphone (
	model		 VARCHAR(512) NOT NULL,
	operativesystem VARCHAR(512) NOT NULL,
	product_id	 BIGINT,
	product_version BIGINT,
	PRIMARY KEY(product_id,product_version)
);

CREATE TABLE notification (
	id		 BIGINT,
	title	 VARCHAR(512) NOT NULL,
	description VARCHAR(512),
	PRIMARY KEY(id)
);

CREATE TABLE commentnotification_comment (
	comment_id	 BIGINT UNIQUE NOT NULL,
	comment_text	 TEXT(512) NOT NULL,
	comment_parentid BIGINT,
	user_id		 BIGINT NOT NULL,
	product_id	 BIGINT NOT NULL,
	product_version	 BIGINT NOT NULL,
	notification_id	 BIGINT,
	PRIMARY KEY(notification_id)
);

CREATE TABLE productnotification_item (
	item_quantity	 INTEGER NOT NULL,
	item_product_id	 BIGINT UNIQUE NOT NULL,
	item_product_version BIGINT UNIQUE NOT NULL,
	notification_id	 BIGINT,
	PRIMARY KEY(notification_id)
);

CREATE TABLE seller (
	companyname VARCHAR(512) NOT NULL,
	user_id	 BIGINT,
	PRIMARY KEY(user_id)
);

CREATE TABLE buyer (
	user_id BIGINT,
	PRIMARY KEY(user_id)
);

CREATE TABLE order (
	id		 BIGINT,
	ordertimestamp TIMESTAMP NOT NULL,
	buyer_user_id	 BIGINT NOT NULL,
	PRIMARY KEY(id)
);

CREATE TABLE classification (
	rating					 BIGINT,
	comment					 VARCHAR(512),
	productnotification_item_notification_id BIGINT UNIQUE NOT NULL
);

CREATE TABLE admin (
	user_id BIGINT,
	PRIMARY KEY(user_id)
);

CREATE TABLE productnotification_item_order (
	productnotification_item_notification_id BIGINT,
	order_id				 BIGINT NOT NULL,
	PRIMARY KEY(productnotification_item_notification_id)
);
